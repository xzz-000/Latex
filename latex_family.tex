\newcommand{\texfamilylogos}{%
  \begin{center}
    \scalebox{2}{%
      \begin{tabular}{ll}
        \hline
        \textbf{标志} & \textbf{命令} \\
        \hline
        \TeX          & \texttt{\textbackslash TeX}       \\
        \LaTeX        & \texttt{\textbackslash LaTeX}     \\
        \LaTeXe       & \texttt{\textbackslash LaTeXe}    \\
        \XeTeX        & \texttt{\textbackslash XeTeX}     \\
        \LuaTeX       & \texttt{\textbackslash LuaTeX}    \\
        \BibTeX       & \texttt{\textbackslash BibTeX}    \\
        \AmS-\TeX     & \texttt{\textbackslash AmS-\TeX}  \\
        \textsl{ConTeXt} & \texttt{\textbackslash textsl\{ConTeXt\}} \\ % 需要 holtxdoc 宏包
        \hline
      \end{tabular}
    }
  \end{center}
}
